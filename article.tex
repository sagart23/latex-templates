\begin{theorem}{Pythagorean Theorem}
In a right-angled triangle, the square of the hypotenuse equals the sum of squares of the other two sides:
\[ a^2 + b^2 = c^2 \]
\end{theorem}

\begin{example}{Computing Area}
Find the area of a circle with radius \SI{5}{cm}.

\begin{Mysolutionbox}
Using $A = \pi r^2$:
\[ A = \pi \times (5)^2 = 25\pi \approx \SI{78.54}{cm^2} \]
\end{Mysolutionbox}
\end{example}
